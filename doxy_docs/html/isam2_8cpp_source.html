167 void t_associate(vector<Point2> *cone_obs, vector<Pose2> *global_obs_cones,
168 vector<Pose2> *all_cone_est, Pose2 global_odom, vector<float> *m_dist,
169 int lo, int hi)
170 {
171 int i = lo;
172 int landmark_idx = lo % (n_landmarks + 1);
173 int obs_id = (int)(lo / (n_landmarks + 1));
174 //RCLCPP_INFO(logger, "lo: %d | hi: %d", lo, hi);
175
176 while (i < hi && obs_id < cone_obs->size())
177 {
178
179 double bearing = std::atan2(cone_obs->at(obs_id).y(),
180 cone_obs->at(obs_id).x());
181
182 double range = norm2(cone_obs->at(obs_id));
183
184 double global_cone_x = (global_odom.x() + range *
185 cos(bearing + global_odom.theta()));
186
187 double global_cone_y = (global_odom.y() + range *
188 sin(bearing + global_odom.theta()));
189 /* TODO: "valid bit" for whether global pose has already been calculated at obs_id */
190
191 //TODO: move this outside of t_associate, and before threads are spawned
192 global_obs_cones->at(obs_id) = Pose2(global_cone_x, global_cone_y, bearing);
193
194
208 int last_est_for_cur_obs = (obs_id + 1) * (n_landmarks + 1) - 1;
209
210
211 /* calculating the mahalanobis distances */
212 for (; i < last_est_for_cur_obs && i < hi; i++)
213 {
218 Eigen::MatrixXd diff(1, 3);
219 Pose2 prev_est = all_cone_est->at(landmark_idx);
220 diff << global_cone_x - prev_est.x(),
221 global_cone_y - prev_est.y(),
222 1;
223
224 /* calculate the mahalanobis distance */
225 m_dist->at(i) = (diff * isam2.marginalCovariance(L(landmark_idx)) * diff.transpose())(0, 0);
226
227 landmark_idx++;
228 }
229
230 if (i == hi)
231 {
232 return;
233 }
234
235 m_dist->at(last_est_for_cur_obs) = M_DIST_TH;
236 i++; /* skip 1 element (reserved for M_DIST_TH) */
237 obs_id++;
238
239 /* new multiple of (n_landmarks+1) means calculating for previous cone/landmark 0 */
240 landmark_idx = 0;
241 }
242
243 }
